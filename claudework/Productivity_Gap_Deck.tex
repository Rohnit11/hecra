% ============================================================
%  DEVIL'S ADVOCATE
% ============================================================

\begin{frame}{A skeptic says: the macro estimate is too low, the J-curve is coming}
\textbf{The objection.}\quad
The 1.56\% ceiling counts only existing tasks at today's $\sim$14\% cost savings. Goldman's 7\% and McKinsey's 1.5\textendash 3.4 pp/yr could be right if AI generates \alert{new tasks} the rubric does not see.\\[0.3cm]
\textbf{The response.}
\begin{itemize}
\item \alert{Adoption is a headwind, not a tailwind.} Only 1.5\% of US firms had invested in AI by 2019 \textemdash{} 10-year cost reductions are likely \emph{over}stated.
\item \alert{New tasks have a sign problem.} Acemoglu's bad-task calibration: AI-enabled social media $+$ IT-security $\approx$ 2\% of GDP, ratio of harm-to-benefit $-19/53$. Apparent $+$2\% GDP coexists with $-$0.72\% welfare.
\item \alert{Micro converges with macro.} Demirci's 21\% drop and Goldberg's 29\% artist exit are \emph{narrow} \textemdash{} exactly what Hulten predicts when 4.6\% of GDP is exposed at $\bar\pi\!\approx\!14\%$.
\end{itemize}
\end{frame}
