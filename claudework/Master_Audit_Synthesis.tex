\subsection{Veneer 1: Speed that hides errors}

\begin{frame}{LLM search halved task time \textemdash{} and halved accuracy on the hard task}
\begin{center}
\renewcommand{\arraystretch}{1.15}
\begin{tabular}{@{}>{\columncolor{LightGray}}p{4.6cm}p{3.6cm}p{3.6cm}@{}}
\toprule
\textbf{Outcome} & \textbf{Traditional search} & \textbf{LLM-based search} \\
\midrule
Median task duration & 3.4 min & \alert{1.6 min} \,($p\!<\!.001$) \\
Easy-task accuracy & 92.3\% & 95.3\% \,(NS) \\
\textbf{Hard-task accuracy} & \textbf{93\%} & \alert{\textbf{47\%}} \,($p\!<\!.001$) \\

\subsection{Veneer 4: Utility that overrides distrust}

\transitionslide{Veneer 4: utility that overrides distrust}{Users adopt what they doubt \textemdash{} if it is convenient.}

% ---------- WARDLE: UTILITY OVER TRUST ----------
\begin{frame}{56\% of users adopt AI for health information \emph{despite} distrusting it}
\vspace{0.1cm}
\begin{itemize}
\item \textbf{15/27 (56\%)} used AI-generated health info despite expressing skepticism, when it ``helped power further investigation.''
\item \textbf{14/27 (52\%)} already integrate ChatGPT into their health-search routines; \textbf{23/27 (85\%)} reported favorably on its formatting or efficiency.
\item AI is used as a \alert{``starting point''}: keyword generation, summarization, then verified on Google or with a doctor \textemdash{} \emph{utility}, not credibility, drives integration.
\item Only \alert{5/27 (19\%)} understood that training data shapes LLM outputs.
\item \textbf{2/3} expressed concern that ChatGPT does not cite sources \textemdash{} yet still used it.
\end{itemize}
\vspace{0.2cm}
\begin{flushright}
\footnotesize\textcolor{WarmGray}{Wardle, Urbani \& Wang (2025), think-aloud protocol, $N\!=\!27$, JMIR 27:e79961.}
\end{flushright}
\end{frame}

\subsection{Devil's advocate \& synthesis}

% ---------- DEVIL'S ADVOCATE ----------
\begin{frame}{A skeptic would say these are lab effects, not real-world behavior}
\vspace{0.1cm}
\textbf{The objection.} Spatharioti uses MTurk ($N\!=\!90$, no demographics); Sharma uses Prolific ($N\!=\!115$\textendash 223, closed-world docs); Wardle is qualitative, $N\!=\!27$ from four East Coast cities. Only Li \& Aral has scale, and even that is a controlled vignette.\\[0.4cm]
\textbf{The response.}
\begin{itemize}
\item \alert{The effect sizes are large.} Cohen's $D\!>\!1$ in Sharma; a 46-point accuracy gap in Spatharioti; $\beta$'s significant at $p\!<\!.001$ in Li \& Aral. Small samples find them anyway.
\item \alert{Four very different methods converge.} Between-subjects RCT, factorial RCT, large pre-registered RCT, qualitative think-aloud \textemdash{} all four point to interface-driven effects exceeding content-driven ones.
\item \alert{The lab is the world.} Li \& Aral's exposure audit: \textbf{42\%} of US Google searches now return AI Overviews; \textbf{56\%} for general-knowledge queries. The vignettes describe what most users now see.
\end{itemize}
\end{frame}

% ---------- SYNTHESIS ----------
\begin{frame}{Restoring user judgment requires interface-level, not model-level, fixes}
\begin{center}

\toprule
\textbf{Veneer} & \textbf{Reform} & \textbf{Concrete evidence} \\
\midrule
Speed & Confidence highlighting on uncertain tokens & Spatharioti Exp.~2: 26\%$\to$58\% accuracy \\
References & Verified-source badges; degrade UI on broken links & Li \& Aral: refs trusted regardless of validity \\

\bottomrule
\end{tabular}
\end{center}

\footnotesize\textcolor{WarmGray}{Each fix is interface-side; none requires changing the underlying model.}
\end{flushright}
\end{frame}

\subsection{Devil's advocate \& synthesis}

% ---------- DEVIL'S ADVOCATE ----------
\begin{frame}{Devil's advocate: aren't most queries the kind where citations don't matter?}
\begin{block}{The strongest objection}
For ``what time is the show,'' ``capital of Peru,'' ``recipe for risotto,'' AI answers users more cheaply than ten blue links \textemdash{} and a wrong citation costs nothing.
\end{block}
\textbf{Reply.} Three problems with the carve-out:
\begin{itemize}
\item Liu (2023): \(r = -0.96\) between fluency and citation precision \textemdash{} users have \emph{no signal} that flips them from ``trust'' to ``verify.''
\item Jazwinska (2025): premium-tier engines decline less, so confident wrongness rises with willingness to pay.
\item Khosravi (2026) + Larsen (2025): even ``trivial'' queries strip the long-tail traffic that funds the verifiable corpus.
\end{itemize}
The cheap query subsidises the expensive one. Sever the cross-subsidy, both fail.
\end{frame}

\subsection{Theory \& reform}

\transitionslide{Theory: naive responses backfire}{The obvious fix \textemdash{} ``just allow more training'' \textemdash{} can destroy welfare.}

% ---------- YANG: SCARCE REGIME ----------
\begin{frame}{When data is scarce, generous fair use \alert{harms} AI development}
\vspace{0.2cm}
\begin{itemize}
\item \textbf{Generous fair use} ($f \to 0$): kills creator incentives $\rightarrow$ \alert{no human content} for training $\rightarrow$ model improvement stalls. \emph{(Prop.~10)}
\item \textbf{Strict fair use} ($f$ too large): the AI company \alert{abandons model improvement} entirely. \emph{(Prop.~11)}
\item \textbf{A moderate $f$} maximizes model quality $X_2$, social welfare, and consumer surplus.
\end{itemize}
\vspace{0.4cm}
\begin{block}{The dynamic problem}

\end{itemize}

$X_2$ is \alert{inverted-U} in $\phi$ when data is scarce \emph{(Prop.~12)}. The two policies cannot be designed independently.\\[0.3cm]
\begin{flushright}
\footnotesize\textcolor{WarmGray}{Yang \& Zhang (2024), Figure 9 \& Proposition 12.}
\end{flushright}
\end{frame}

% ---------- DETECTION ----------
\begin{frame}{Disclosure cannot enforce the AI / non-AI line \textemdash{} and creates perverse incentives}
\vspace{0.3cm}
\begin{itemize}
\item \textbf{The ``Old Hard Drive Debacle''} \emph{(Cooper 2025)}: artists must log every GenAI use \alert{solely to strip themselves of rights} $\rightarrow$ strong incentive to \emph{deny} AI use.
\item \textbf{No robust detection tool} exists for whether \textemdash{} let alone \emph{which elements} of \textemdash{} a work used GenAI.\\
{\footnotesize\textcolor{WarmGray}{(10+ sources on watermarking limits cited in Cooper 2025)}}
\item \textbf{Convergence problem:} search-engine and AI-training crawlers are technologically indistinguishable; blocking one blocks the other.
\end{itemize}

% ---------- INFRINGEMENT + SYNTHETIC ----------
\begin{frame}{Synthetic data is no escape \textemdash{} model collapse and copy-mining await}
\vspace{0.2cm}
\begin{itemize}
\item GenAI produces \alert{near-identical copies} of protected works even with developer safeguards.\\
{\footnotesize\textcolor{WarmGray}{(Cooper \& Grimmelmann 2025)}}
\item US ``Blurred Lines'' precedent: \alert{weak similarity thresholds} mean outputs need not be close replicas to infringe.
\item \textbf{Copy-mining incentive}: mass-generate to claim maximum rights, mining out the latent space between existing works.\\
{\footnotesize\textcolor{WarmGray}{(Cooper 2025; Lemley 2024)}}
\item \textbf{Synthetic data fails as a fix:} depends on real-data quality, cannot generate genuine novelty, recursive training risks \alert{model collapse}.\\
{\footnotesize\textcolor{WarmGray}{(Shumailov et al.\ 2024)}}
\end{itemize}
\end{frame}

% ---------- DEVIL'S ADVOCATE ----------
\begin{frame}{A skeptic would say the model is too stylized to drive real policy}
\vspace{0.2cm}
\textbf{The objection.} Yang \& Zhang assume uniform skill, competitive AI markets, binary content quality, and a single regulator. Cooper et al.\ provide no original empirical evidence.\\[0.4cm]
\textbf{The response.}
\begin{itemize}
\item Yang \& Zhang's \alert{regime-dependence result} (abundant vs.\ scarce) is robust across all numerical examples and does not hinge on the uniform-skill closed form.
\item The Cooper primer's job is not estimation \textemdash{} it identifies \alert{empirical gaps} (representativeness of opt-out data, creator-behavior responses, output memorization rates) for the next generation of work.
\item Both papers converge on the same conclusion from independent methods: \alert{copyright doctrine cannot be optimized one lever at a time, in one jurisdiction, at one moment in time.}
\end{itemize}
\end{frame}
