% ---------- JAGADEESAN: SETUP ----------
\begin{frame}{Two firms, two objectives \textemdash{} entrant exploits a weaker safety threshold}
\begin{center}
\renewcommand{\arraystretch}{1.0}
\begin{tabular}{@{}>{\columncolor{LightGray}}p{2.8cm}>{\RaggedRight\arraybackslash}p{8.8cm}@{}}
\toprule
\textbf{Object} & \textbf{Definition} \\
\midrule
$\alpha \in [0.5,1]$ & Share of data labeled with performance (vs.\ safety) objective \\
$\rho \in [0,1)$ & Correlation between $\beta_1$ (perf) and $\beta_2$ (safety) regressors \\
$\tau_I, \tau_E$ & Safety constraints; \alert{$\tau_E > \tau_I$} (regulatory asymmetry) \\
$\nu = 0.34$ & Single-objective scaling exponent, calibrated to Chinchilla \\
$N_E^*$ & Min.\ entrant data to match incumbent at $\tau_E$ \\
\bottomrule
\end{tabular}
\end{center}
\textbf{Theorem 1.} Entrant can enter with \emph{finite} data even at $N_I\!=\!\infty$, if $\tau_I$ binds and $\rho<1$.
\begin{flushright}
\footnotesize\textcolor{WarmGray}{Jagadeesan, Jordan \& Steinhardt (2025), \emph{PNAS}; pure theory, no data.}
\end{flushright}
\end{frame}

% ---------- JAGADEESAN: SCALING REGIMES ----------
\begin{frame}{Multi-objective scaling exponent collapses as data grows: $0.34\!\to\!0.25\!\to\!0$}
\begin{center}
\renewcommand{\arraystretch}{1.1}
\begin{tabular}{@{}>{\columncolor{LightGray}}p{1.4cm}>{\RaggedRight\arraybackslash}p{4.6cm}>{\RaggedRight\arraybackslash}p{2.4cm}>{\RaggedRight\arraybackslash}p{2.6cm}@{}}
\toprule
\textbf{Regime} & \textbf{Range of $N$} & \textbf{$\nu^*$ (Thm 2)} & \textbf{Calibrated} \\
\midrule
1 & $N \leq (1\!-\!\alpha)^{-1/\nu}(1\!-\!\rho)^{-1/\nu}$ & $\nu$ & $0.34$ \\
2 & intermediate & $\nu/(\nu\!+\!1)$ & $0.25$ \\
3 & $N \geq (1\!-\!\alpha)^{-(2+\nu)/\nu}(1\!-\!\rho)^{-1/\nu}$ & $\mathbf{0}$ & \alert{$0$} \\
\bottomrule
\end{tabular}
\end{center}
\vspace{0.1cm}
\textbf{Theorem 4 corollary.} $N_E^* = \Theta(N_I^c)$, where $c\!=\!1\!\to\!0.75\!\to\!0$ as $N_I\!\to\!\infty$.
Past a threshold, more incumbent data buys nothing the entrant can't match cheaply.
\begin{flushright}
\footnotesize\textcolor{WarmGray}{Jagadeesan et al.\ (2025), Theorems~2, 4; calibration to Hoffmann et al.\ (2022) Chinchilla scaling.}
\end{flushright}
\end{frame}

% ---------- JAGADEESAN: CATCH ----------
\begin{frame}{The mechanism that lowers barriers also tempts a race-to-the-bottom on safety}
\begin{center}
\renewcommand{\arraystretch}{1.1}
\begin{tabular}{@{}>{\columncolor{LightGray}}p{4.0cm}>{\RaggedRight\arraybackslash}p{7.6cm}@{}}
\toprule
\textbf{Theorem 5 finding} & \textbf{Implication} \\
\midrule
$N_E^*\!=\!\Theta(D^{-c})$, $D\!=\!G_I\!-\!G_E$ & As safety thresholds converge, entry threshold \emph{explodes} \\
$c$ rises from $1/\nu$ ($\!=\!2.94$) to $(\nu'\!+\!1)/\nu'$ as $D\!\downarrow$ & Entrant must dataset-scale ever harder \\
\bottomrule
\end{tabular}
\end{center}
\vspace{0.1cm}
\textbf{Policy corner.} Stricter incumbent-only safety regulation (DSA's VLOP $\geq$ 45M users; FTC investigations) \emph{lowers} entry barriers \textemdash{} but only if entrants' weaker safety is socially tolerable. Otherwise the gap closes and entry collapses.
\begin{flushright}
\footnotesize\textcolor{WarmGray}{Jagadeesan et al.\ (2025), Theorem~5; Section ``Implications,'' p.\ 9.}
\end{flushright}
\end{frame}

% ---------- DEVIL'S ADVOCATE ----------
\begin{frame}{Devil's advocate: aren't all three findings just early-stage snapshots?}
\begin{block}{The strongest objection}
Carugati (2023) is pre-DeepSeek; Jagadeesan (2025) has \emph{no data}; Zhang (2024) is 5 months, one platform, $N=126$. The frontier might be contestable today and concentrated tomorrow.
\end{block}
\textbf{Reply.} The synthesis is not that concentration won't happen; it is that \emph{whether} it happens depends on the policy choices each lens names:
\begin{itemize}
\item Carugati: regulators must protect open-source from over-restriction.
\item Jagadeesan: $\tau_E>\tau_I$ is the lever; Theorem 5 shows converging $\tau$s collapse entry.
\item Zhang: low-end expansion is real, but supplier surplus depends on platform fees.
\end{itemize}
The frontier is contestable \emph{conditional on} these levers. Pull them wrong, the market tips.
\end{frame}

The same three levers work either way. The market does not concentrate by physics; it concentrates by choice.
