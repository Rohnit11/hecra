% ---------- IDE: SETUP ----------
\begin{frame}{Garicano hierarchies + compute: AI agents at exogenous knowledge $z_{AI}\in[0,1)$}
\begin{center}
\renewcommand{\arraystretch}{1.0}
\begin{tabular}{@{}>{\columncolor{LightGray}}p{2.6cm}>{\RaggedRight\arraybackslash}p{8.8cm}@{}}
\toprule
\textbf{Object} & \textbf{Definition} \\
\midrule
$z\sim G$ on $[0,1]$ & Human knowledge; partition into $W$ (workers), $I$ (independents), $S$ (solvers) \\
$h<h_0$ & Solver time per question; admits two-layer firms \\
$m$ (compute) & Abundant; 1 unit $\to$ 1 AI agent of fixed knowledge $z_{AI}$ \\
Autonomous AI & Agent acts as worker AND solver; nonautonomous = copilot only \\
\bottomrule
\end{tabular}
\end{center}
\textbf{Prop 2.} $r^*\!=\!z_{AI}$. If $z_{AI}\!\in\!\text{int}\,W$, basic AI; if $z_{AI}\!\in\!\text{int}\,S$, advanced AI \textemdash{} displacement direction flips.
\begin{flushright}
\footnotesize\textcolor{WarmGray}{Ide \& Talam\`{a}s (2025), \emph{JPE}; pure theory; Props.\ 1\textendash 2.}
\end{flushright}
\end{frame}

% ---------- IDE: PROP 5 WINNERS AT EXTREMES ----------
\begin{frame}{Proposition 5: winners at the extremes; the individual at $z_{AI}$ always loses}
\begin{center}
\renewcommand{\arraystretch}{1.1}
\begin{tabular}{@{}>{\columncolor{LightGray}}p{3.6cm}>{\RaggedRight\arraybackslash}p{8.0cm}@{}}
\toprule
\textbf{Set} & \textbf{When non-empty} \\
\midrule
$T=(z_t,1]$ (top winners) & \textcolor{Teal}{\textbf{Always}} non-empty under $h<h_0$, $z_{AI}<1$ \\
$B=[0,z_b)$ (bottom winners) & Non-empty \alert{iff} $z_{AI}>\bar z_{AI}\in\text{int}\,W$ \\
$\{z_{AI}\}$ (pivot individual) & \alert{$w^*(z_{AI})<w(z_{AI})$} \emph{always} \\
\bottomrule
\end{tabular}
\end{center}
\vspace{0.1cm}
\textbf{The shape of the welfare curve is a U with a cliff at $z_{AI}$.} The least knowledgeable gain only when AI is capable enough; the most knowledgeable always gain; the human who does exactly what the AI does is uniformly displaced.
\begin{flushright}
\footnotesize\textcolor{WarmGray}{Ide \& Talam\`{a}s (2025), Proposition~5, p.~3788.}
\end{flushright}
\end{frame}

\begin{center}
\renewcommand{\arraystretch}{1.0}
\begin{tabular}{@{}>{\columncolor{LightGray}}p{3.4cm}>{\RaggedRight\arraybackslash}p{4.0cm}>{\RaggedRight\arraybackslash}p{4.4cm}@{}}
\toprule
\textbf{Regime} & \textbf{Best for least-knowledgeable} & \textbf{Best for most-knowledgeable} \\
\midrule
Autonomous AI & only if $z_{AI}>\bar z_{AI}$ & \textcolor{Teal}{always} \\
Nonautonomous AI & \textcolor{Teal}{always}, if $z_{AI}>w(0)$ & dominated by autonomous \\
No AI & \textemdash & \textemdash \\
\bottomrule
\end{tabular}
\end{center}
\textbf{Reconciles micro evidence:} copilot RCTs (Brynjolfsson--Li--Raymond 2025) map to nonautonomous $\to$ help the bottom. Coworker deployments (Klarna; Alekseeva 2024) map to basic autonomous $\to$ displace the bottom. Same AI is ``basic'' in HIC and ``advanced'' in LIC \textemdash{} direction flips by economy.
\begin{flushright}
\footnotesize\textcolor{WarmGray}{Ide \& Talam\`{a}s (2025), Prop.~6; \S VII, pp.~3791\textendash 3796.}
\end{flushright}
\end{frame}

% ---------- DEVIL'S ADVOCATE ----------
\begin{frame}{Devil's advocate: aren't these three lenses measuring different things at different times?}
\begin{block}{The strongest objection}
Padilla is desktop-only, US-skewed, $N\!=\!2{,}041$. Liu is web-traffic only \textemdash{} no API, mobile, or enterprise. Ide is pure theory with $z_{AI}<1$. Maybe the asymmetries are artifacts of measurement.
\end{block}
\textbf{Reply.} The lenses converge because they measure complementary slices of one substitution:
\begin{itemize}
\item Padilla's bottom-25\% sites and education $=$ Ide's $z<z_{AI}$ long-tail informational supply.
\item Liu's $0.6$\textendash$0.8$ log-GDP elasticity says $z_{AI}$'s relative position is country-specific.
\item Ide's pivot at $z_{AI}$ explains why Stack Overflow falls $-68.8\%$ but Reddit does not.
\end{itemize}
The asymmetry is structural, not measurement noise.
\end{frame}

% ---------- SYNTHESIS ----------
\begin{frame}{The diffusion is lopsided in three layered ways \textemdash{} they compound, they do not cancel}
\begin{center}
\renewcommand{\arraystretch}{1.05}
\begin{tabular}{@{}>{\columncolor{LightGray}}p{2.4cm}>{\RaggedRight\arraybackslash}p{4.4cm}>{\RaggedRight\arraybackslash}p{5.4cm}@{}}
\toprule
\textbf{Layer} & \textbf{Asymmetry} & \textbf{Compound effect} \\
\midrule
\textbf{Behavior} & Long-tail informational supply substituted, social web spared & Site economics flip for the bottom 4.5M websites \\
\textbf{Geography} & HIC adopt; LIC do not & Where $z_{AI}$ sits in the local skill distribution differs by country \\
\textbf{Theory} & Welfare U-shape with cliff at $z_{AI}$ & The cliff lands on different occupations in HIC vs.\ LIC \\
\bottomrule
\end{tabular}
\end{center}
\vspace{0.1cm}
The same AI cannot be expected to produce the same outcomes across users, countries, or skill bands. Aggregation hides the cliff.
\end{frame}
